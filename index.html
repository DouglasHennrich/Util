var fb = Alloy.Globals.FB;

/*
@ Refresh Permissions
*/
fb.addEventListener('tokenUpdated', function(e) {
    console.log('Updated permissions: ' + JSON.stringify(fb.permissions));
});

/*
@ Login Listener
*/
fb.addEventListener('login',function(event){
    if( event.success ){
        console.log('login data: ', event.data);
        console.log('\n===================\n');
        console.log('login UID: ', event.uid);

        // Refresh Permissions if changed
        fb.refreshPermissionsFromServer();

        //
        goMain(JSON.parse(event.data));

    }else{
        console.log('Login Fail: ', event.error);
    }
});

/*
@ goMain
*/
function goMain(_user){
    Alloy.createController('main',{
        user : _user
    })
     .getView()
      .open();
};
